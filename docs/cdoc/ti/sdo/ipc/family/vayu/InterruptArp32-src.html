    40    import ti.sdo.utils.MultiProc;
    41    
    42    /*!
    43     *  ======== InterruptArp32 ======== 
    44     *  ARP32 based interrupt manager
    45     */
    46    
    47    module InterruptArp32 inherits ti.sdo.ipc.notifyDrivers.IInterrupt
    48    {
    49        /* Base address for the Mailbox subsystem */
    50        config UInt32 mailboxBaseAddr[10];
    51    
    52        /* TODO: Document how this table is generated */
    53        config UInt32 mailboxTable[64];
    54    
    55        config UInt32 eveInterruptTable[8];
    56    
    57        config UInt32 procIdTable[8];
    58    
    59    internal:
    60    
    61        /*! Statically retrieve procIds to avoid doing this at runtime */    
    62        config UInt eve0ProcId   = MultiProc.INVALIDID;
    63        config UInt eve1ProcId   = MultiProc.INVALIDID;
    64        config UInt eve2ProcId   = MultiProc.INVALIDID;
    65        config UInt eve3ProcId   = MultiProc.INVALIDID;
    66        config UInt dsp0ProcId   = MultiProc.INVALIDID;
    67        config UInt dsp1ProcId   = MultiProc.INVALIDID;
    68        config UInt videoProcId  = MultiProc.INVALIDID;
    69        config UInt vpssProcId   = MultiProc.INVALIDID;
    70    
    71        /*! Function table */
    72        struct FxnTable {
    73            Fxn    func;
    74            UArg   arg;
    75        }
    76    
    77        /*! Stub to be plugged for dsp-arp32 interrupts */
    78        Void intShmStub(UArg arg);
    79        
    80        struct Module_State {        
    81            /* 
    82             * Create a function table of length 8 (Total number of cores in the
    83             * System) for each EVE core.
    84             */
    85            FxnTable   fxnTable[8];
    86            UInt       numPlugged[10];  /* # of times the interrupt was registered */
    87        };
    88    }
